import numpy as np
import matplotlib.pyplot as plt

# ===============================
# PARAMETERS AND CONSTANTS
# ===============================
K = 20  # Thermal conductivity (W/m.K)
RHO = 800  # Density (kg/m^3)
CP = 2500  # Specific heat capacity (J/kg.K)
H = 100  # Convection coefficient (W/m^2.K)
SIGMA = 5.670374e-8  # Stefan-Boltzmann constant (W/m^2.K^4)
EPSILON = 0.2  # Surface emissivity

HEIGHT = 0.1  # Height of the candle (m)
DIAMETER = 0.03  # Diameter of the candle (m)

T_AMBIENT_INITIAL = 300.0  # Initial ambient temperature (K)
T_AMBIENT_FINAL = 223.0  # Final ambient temperature (K)
T_FLAME = 1800.0  # Flame temperature (K)

DR = 0.001  # Radial step size (m)
DZ = 0.002  # Axial step size (m)
DT = 0.01  # Time step size (s)

R_MAX = DIAMETER / 2
Z_MAX = HEIGHT
NR = int(R_MAX / DR) + 1
NZ = int(Z_MAX / DZ) + 1

# ===============================
# FUNCTIONS
# ===============================

def initialize_temperature_grid(T_initial=T_AMBIENT_INITIAL):
    """
    Initializes the temperature grid with the ambient initial temperature.
    """
    return np.full((NZ, NR), T_initial, dtype=np.float64)


def apply_boundary_conditions(T, T_flame=T_FLAME, T_ambient=T_AMBIENT_FINAL):
    """
    Applies boundary conditions to the temperature grid.
    """
    T[-1, :] += DT * EPSILON * SIGMA * (T_flame**4 - T[-1, :]**4)
    T[:, -1] -= DT * H * (T[:, -1] - T_ambient)
    T[0, :] = T[1, :]  # Insulated base
    return T


def solve_heat_equation(T, timesteps=100):
    """
    Solves the heat equation using the explicit finite difference method.
    """
    for t in range(timesteps):
        T_new = np.copy(T)
        for z in range(1, NZ - 1):
            for r in range(1, NR - 1):
                dT_dr2 = (T[z, r + 1] - 2 * T[z, r] + T[z, r - 1]) / (DR**2)
                dT_dz2 = (T[z + 1, r] - 2 * T[z, r] + T[z - 1, r]) / (DZ**2)
                T_new[z, r] += DT * (K / (RHO * CP)) * (dT_dr2 + dT_dz2)

        T = apply_boundary_conditions(T_new)

        # Debugging: Check evolution of the grid
        if t % 10 == 0:
            print(f"Time step {t}: Min={T.min():.2f}, Max={T.max():.2f}")
    return T


def analyze_results(T, title="Temperature Profiles"):
    """
    Analyzes the results and generates plots.
    """
    heights = [0, NZ // 2, NZ - 1]
    plt.figure(figsize=(10, 6))
    for z in heights:
        plt.plot(np.linspace(0, R_MAX, NR), T[z, :], label=f"Height {z * DZ:.3f} m")

    plt.title(title)
    plt.xlabel("Radius (m)")
    plt.ylabel("Temperature (K)")
    plt.legend()
    plt.grid()
    plt.show()


# ===============================
# MAIN FUNCTION
# ===============================
def main():
    """
    Main function to simulate the candle heat transfer.
    """
    T = initialize_temperature_grid()

    # Phase 1: Initial steady state
    T = solve_heat_equation(T, timesteps=100)
    analyze_results(T, title="Temperature Profiles at 300 K")

    # Phase 2: Transition to 223 K
    T = apply_boundary_conditions(T, T_ambient=T_AMBIENT_FINAL)
    T = solve_heat_equation(T, timesteps=100)
    analyze_results(T, title="Temperature Profiles at 223 K")


if __name__ == "__main__":
    main()

import numpy as np
import matplotlib.pyplot as plt

def plot_heatmap(T, R_MAX, NR, DR, title="Heatmap of Temperature Distribution", z_level=0):
    """
    Plots the temperature distribution as a heatmap at a specific z-level.
    Args:
        T: 3D numpy array of temperature values (z, r)
        R_MAX: Maximum radius
        NR: Number of radial points
        DR: Radial step size
        title: Title of the plot
        z_level: The z-index to plot
    """
    # Create radial grid
    r = np.linspace(-R_MAX, R_MAX, 2 * NR)  # Extend radius symmetrically
    x, y = np.meshgrid(r, r)
    radial_distances = np.sqrt(x**2 + y**2)

    # Create the temperature matrix for the full circle
    full_circle_temperature = np.zeros_like(x)
    for i in range(full_circle_temperature.shape[0]):
        for j in range(full_circle_temperature.shape[1]):
            r_idx = int(abs(radial_distances[i, j] / DR))
            if r_idx < NR:  # Valid radius
                full_circle_temperature[i, j] = T[z_level, r_idx]

    # Create the plot
    plt.figure(figsize=(8, 6))
    heatmap = plt.imshow(
        full_circle_temperature,
        extent=[-R_MAX, R_MAX, -R_MAX, R_MAX],
        origin="lower",
        cmap="hot",
        interpolation="nearest",
        vmin=np.min(T),
        vmax=np.max(T),
    )
    plt.colorbar(heatmap, label="Temperature (K)")
    plt.title(title)
    plt.xlabel("X (m)")
    plt.ylabel("Y (m)")
    plt.grid(False)
    plt.show()

# Example usage:
# Define parameters
R_MAX = 1.0  # Maximum radius in meters
NR = 50      # Number of radial points
DR = R_MAX/NR  # Radial step size

# Create sample temperature data (3D array: z, r)
# For example, a simple temperature distribution that decreases with radius
z_levels = 10
T = np.zeros((z_levels, NR))
for z in range(z_levels):
    T[z, :] = 300 - 50 * np.linspace(0, 1, NR)  # Temperature decreases from 300K to 250K

# Plot the heatmap for z_level = 0
plot_heatmap(T, R_MAX, NR, DR, title="Temperature Distribution", z_level=0)

import numpy as np
import matplotlib.pyplot as plt

# Generate sample data (customize this for your data)
x = np.linspace(-1, 1, 100)  # Spatial domain along x-axis
y = np.linspace(0, 1, 50)   # Spatial domain along y-axis
X, Y = np.meshgrid(x, y)

# Peclet numbers
peclet_numbers = [10, 100, 1000]
concentration_profiles = [np.exp(-Pe * (X**2 + Y)) for Pe in peclet_numbers]

# Plotting
fig, axes = plt.subplots(len(peclet_numbers), 1, figsize=(8, 8), sharex=True)
for i, (Pe, concentration) in enumerate(zip(peclet_numbers, concentration_profiles)):
    im = axes[i].imshow(
        concentration,
        extent=[x.min(), x.max(), y.min(), y.max()],
        origin='lower',
        cmap='viridis',
        aspect='auto'
    )
    axes[i].set_title(f"Profil de concentration: Pe={Pe}")
    axes[i].set_ylabel("y")
    plt.colorbar(im, ax=axes[i], label="Concentration")

axes[-1].set_xlabel("x")
plt.tight_layout()
plt.show()
